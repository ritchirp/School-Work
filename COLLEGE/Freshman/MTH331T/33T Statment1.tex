\begin{statement}~
Let $n$ be an integer. If $n\geq3$ then $n^3>(n+1)^2$
\end{statement}

\begin{proof}
For $n\geq3$, $n-2\geq1$ and $n+1\geq4$\\
so $n(n-2)(n+1)\geq3\cdot4\cdot1=12>1$\\
$n(n-2)(n+1)>1\\ n^3-n^2-2n>1\\ n^3>n^2+2n+1\\ n^3>(n+1)^2$


\end{proof}
